{%- comment -%} Generic previous/next series navigation, rendered at the bottom of any post that has `series: ` in frontmatter. Reads: - site.data.series[] — series-level metadata - site.data[_series] — ordered list of posts Quietly renders nothing if the post isn't found in the ordered list, so a stray `series: foo` in frontmatter won't blow up the build. {%- endcomment -%} {%- assign series_meta = site.data.series[page.series] -%} {%- assign series_data_key = page.series | append: "_series" -%} {%- assign series_list = site.data[series_data_key] -%} {%- assign current_index = -1 -%} {%- for entry in series_list -%} {%- if entry.url == page.url -%} {%- assign current_index = forloop.index0 -%} {%- break -%} {%- endif -%} {%- endfor -%} {%- if current_index >= 0 -%} {%- assign published_urls = site.posts | where: "series", page.series | map: "url" -%} {%- assign total = series_list.size -%} {%- assign part_num = current_index | plus: 1 -%}
{%- if series_meta.landing_url -%} {{ series_meta.short_title | default: series_meta.title }} {%- else -%} {{ series_meta.short_title | default: series_meta.title }} {%- endif -%} · Part {{ part_num }} of {{ total }}
{%- if current_index > 0 -%} {%- assign prev_index = current_index | minus: 1 -%} {%- assign prev = series_list[prev_index] -%} ← Previous {{ prev.title }} {%- else -%} {%- endif -%} {%- assign next_index = current_index | plus: 1 -%} {%- if next_index < total -%} {%- assign next = series_list[next_index] -%} {%- assign next_published = false -%} {%- for u in published_urls -%} {%- if u == next.url -%}{%- assign next_published = true -%}{%- break -%}{%- endif -%} {%- endfor -%} {%- if next_published -%} Next → {{ next.title }} {%- else -%} Next → {{ next.title }} publishes {{ next.date | date: "%B %-d" }} {%- endif -%} {%- else -%} {%- endif -%}
{%- endif -%}
